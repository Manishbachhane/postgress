-- =========================
-- SQL CLAUSES
-- =========================


-- 1. SELECT
SELECT * FROM employees;

SELECT name, salary FROM employees;


-- 2. FROM
SELECT * FROM employees;


-- 3. WHERE
SELECT * FROM employees
WHERE salary > 50000;


-- 4. ORDER BY
SELECT * FROM employees
ORDER BY salary ASC;

SELECT * FROM employees
ORDER BY salary DESC;


-- 5. GROUP BY
SELECT dept, COUNT(*)
FROM employees
GROUP BY dept;


-- 6. HAVING
SELECT dept, COUNT(*)
FROM employees
GROUP BY dept
HAVING COUNT(*) > 1;


-- 7. LIMIT
SELECT * FROM employees
LIMIT 5;


-- 8. OFFSET
SELECT * FROM employees
LIMIT 5 OFFSET 5;


-- 9. DISTINCT
SELECT DISTINCT dept
FROM employees;


-- 10. AS (Alias)
SELECT name AS employee_name
FROM employees;


-- 11. IN
SELECT * FROM employees
WHERE dept IN ('IT', 'HR');


-- 12. BETWEEN
SELECT * FROM employees
WHERE salary BETWEEN 40000 AND 70000;


-- 13. LIKE
SELECT * FROM employees
WHERE name LIKE 'R%';


-- 14. IS NULL
SELECT * FROM employees
WHERE dept IS NULL;


-- 15. IS NOT NULL
SELECT * FROM employees
WHERE dept IS NOT NULL;


-- 16. AND
SELECT * FROM employees
WHERE dept = 'IT'
AND salary > 50000;


-- 17. OR
SELECT * FROM employees
WHERE dept = 'IT'
OR dept = 'HR';


-- 18. NOT
SELECT * FROM employees
WHERE dept NOT IN ('HR');


-- 19. JOIN
SELECT employees.name, department.dept_name
FROM employees
JOIN department
ON employees.dept_id = department.id;


-- 20. LEFT JOIN
SELECT employees.name, department.dept_name
FROM employees
LEFT JOIN department
ON employees.dept_id = department.id;


-- 21. RIGHT JOIN
SELECT employees.name, department.dept_name
FROM employees
RIGHT JOIN department
ON employees.dept_id = department.id;


-- 22. FULL JOIN
SELECT employees.name, department.dept_name
FROM employees
FULL JOIN department
ON employees.dept_id = department.id;


-- 23. UNION
SELECT name FROM employees
UNION
SELECT name FROM managers;


-- 24. EXISTS
SELECT name
FROM employees e
WHERE EXISTS (
    SELECT 1
    FROM department d
    WHERE d.id = e.dept_id
);


-- 25. CASE
SELECT name,
       CASE
           WHEN salary > 60000 THEN 'High'
           WHEN salary > 40000 THEN 'Medium'
           ELSE 'Low'
       END AS salary_level
FROM employees;
